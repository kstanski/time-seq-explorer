%matplotlib inline

from __future__ import absolute_import
from __future__ import division
from __future__ import print_function

import numpy as np
import matplotlib.pyplot as plt
from sklearn.decomposition import TruncatedSVD


def generate_data_with_mods(training_data, testing_data, seq_length, seed=0):
    np.random.seed(seed)
    no_of_instances = training_data + testing_data
    seqs = list()
    properties = list()
    for _ in range(no_of_instances):
        phase_mod = np.random.choice(2, 1, replace=False)
        #phase_mod = 0.0
        if phase_mod < 0.5:
            phase = np.random.uniform(-1.0, -0.5)
        else:
            phase = np.random.uniform(0.5, 1.0)

        amp_mod = np.random.choice(2, 1, replace=False)
        #amp_mod = 0.0
        if amp_mod < 0.5:
            amplitude = np.random.uniform(1 / 10.0, 1 / 5.0)
        else:
            amplitude = np.random.uniform(1 / 2.0, 1.0)

        freq_mod = np.random.choice(2, 1, replace=False)
        #freq_mod = 0.0
        if freq_mod < 0.5:
            frequency = np.random.uniform(1 / 4.0, 1 / 2.0)
        else:
            frequency = np.random.uniform(1.0, 2.0)

        seq = generate_sequence(seq_length + 1, amplitude, frequency, phase, 0.0)
        seqs.append(seq)
        properties.append([phase, amplitude, frequency])

    # Format dataset
    rand_idxs = np.random.choice(no_of_instances, no_of_instances, replace=False)
    data = [list() for _ in range(4)]
    features_train, features_test, labels_train, labels_test = data
    for i in range(no_of_instances):
        rand_idx = rand_idxs[i]
        features = seqs[rand_idx][:-1]
        #labels = [seqs[rand_idx][-1]]    # Last element of a sequence
        #labels = [properties[rand_idx][1]]    # amplitude
        labels = [properties[rand_idx][2]]    # frequency
        if i < training_data:
            features_train.append(features)
            labels_train.append(labels)
        else:
            features_test.append(features)
            labels_test.append(labels)

    for i, d in enumerate(data):
        data[i] = np.array(d)

    return data


def generate_sequence(length, amplitude, frequency, phase, noise):
    x = np.linspace(0, 10, length)
    y_func = amplitude * np.sin(frequency * x - phase)
    y = y_func + noise * np.random.randn(x.shape[0])
    return y


features_train, features_test, labels_train, labels_test = generate_data_with_mods(1000,
                                                                             1000,
                                                                             50)

data = features_train
PCA_model = TruncatedSVD(n_components=2).fit(data)
pcas = PCA_model.transform(data)
x = pcas[:, 0]
y = pcas[:, 1]

plt.figure(figsize=(12, 10))
ax1 = plt.subplot(1, 1, 1)
ax1.scatter(x, y, s=5)
ax1.set_title('non-interactive PCA train features')
plt.show()
