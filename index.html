from __future__ import absolute_import
from __future__ import division
from __future__ import print_function
import numpy as np


def generate_data_with_mods(training_data, testing_data, seq_length, seed=0):
    np.random.seed(seed)
    no_of_instances = training_data + testing_data
    seqs = list()
    properties = list()
    for _ in range(no_of_instances):
        phase_mod = np.random.choice(2, 1, replace=False)
        #phase_mod = 0.0
        if phase_mod < 0.5:
            phase = np.random.uniform(-1.0, -0.5)
        else:
            phase = np.random.uniform(0.5, 1.0)

        amp_mod = np.random.choice(2, 1, replace=False)
        #amp_mod = 0.0
        if amp_mod < 0.5:
            amplitude = np.random.uniform(1 / 10.0, 1 / 5.0)
        else:
            amplitude = np.random.uniform(1 / 2.0, 1.0)

        freq_mod = np.random.choice(2, 1, replace=False)
        #freq_mod = 0.0
        if freq_mod < 0.5:
            frequency = np.random.uniform(1 / 4.0, 1 / 2.0)
        else:
            frequency = np.random.uniform(1.0, 2.0)

        seq = generate_sequence(seq_length + 1, amplitude, frequency, phase, 0.0)
        seqs.append(seq)
        properties.append([phase, amplitude, frequency])

    # Format dataset
    rand_idxs = np.random.choice(no_of_instances, no_of_instances, replace=False)
    data = [list() for _ in range(4)]
    features_train, features_test, labels_train, labels_test = data
    for i in range(no_of_instances):
        rand_idx = rand_idxs[i]
        features = seqs[rand_idx][:-1]
        #labels = [seqs[rand_idx][-1]]    # Last element of a sequence
        #labels = [properties[rand_idx][1]]    # amplitude
        labels = [properties[rand_idx][2]]    # frequency
        if i < training_data:
            features_train.append(features)
            labels_train.append(labels)
        else:
            features_test.append(features)
            labels_test.append(labels)

    for i, d in enumerate(data):
        data[i] = np.array(d)

    return data


def generate_sequence(length, amplitude, frequency, phase, noise):
    x = np.linspace(0, 10, length)
    y_func = amplitude * np.sin(frequency * x - phase)
    y = y_func + noise * np.random.randn(x.shape[0])
    return y

import plotly
from IPython.core.display import display, HTML
from IPython.display import Javascript
import json

plotly.offline.init_notebook_mode(connected=True)


def draw_js_plot(projection_plot, sequence_plot):
    display(HTML("<div id=" + projection_plot['plot_name'] + "></div>"))
    display(HTML("<div id=" + sequence_plot['plot_name'] + "></div>"))

    js_code = '''
    require(['plotly'], function(Plotly) {

        var scatter_plot = function(element_id, x, y, customdata, title) {
            var data = [{x:x, y:y, customdata:customdata, type:'scatter',
                     mode:'markers', marker:{size:5}}],
                layout = {hovermode:'closest', title:title,
                    width: 500, height: 500};
            Plotly.newPlot(element_id, data, layout);
        };

        var proj_plot_name = ''' + projection_plot['plot_name'] + ''',
            x = ''' + projection_plot['x'] + ''',
            y = ''' + projection_plot['y'] + ''',
            customdata = ''' + projection_plot['customdata'] + ''',
            proj_plot = document.getElementById(proj_plot_name);
        scatter_plot(proj_plot_name, x, y, customdata, 'Projection');

        var seq_plot_name = ''' + sequence_plot['plot_name'] + ''';
        scatter_plot(seq_plot_name, null, null, null, 'Sequence');

        proj_plot.on('plotly_click', function(d) {
            Plotly.animate(seq_plot_name, {
                data: [{y: d.points[0].customdata}],
                traces: [0],
                layout: {hovermode:'closest', title:'Sequence',
                    xaxis: {range: ''' + sequence_plot['x_range'] + '''},
                    yaxis: {range: ''' + sequence_plot['y_range'] + '''}
                }
            },
            {
                transition: {
                  duration: 200,
                  easing: 'cubic-in-out'
                }
            })
        });
    });
    '''
    return js_code
